- [ ] 创建聊天服务 `lib/src/services/chat_service.dart`
  ```dart
  class ChatService {
    late WebSocketChannel _channel;
    final StreamController<ChatMessage> _messageStream = StreamController.broadcast();

    Future<void> connect(String roomId, String token) async {
      _channel = WebSocketChannel.connect(
        Uri.parse('ws://$apiBase/rooms/$roomId/chat?token=$token'),
      );

      _channel.stream.listen(
        (message) {
          final msg = ChatMessage.fromJson(jsonDecode(message));
          _messageStream.add(msg);
        },
        onError: (_) => _reconnect(roomId, token),
        onDone: () => _reconnect(roomId, token),
      );
    }

    void sendMessage(String content) {
      _channel.sink.add(jsonEncode({
        'type': 'text',
        'content': content,
        'timestamp': DateTime.now().toIso8601String(),
      }));
    }

    void dispose() {
      _channel.sink.close();
      _messageStream.close();
    }
  }
  ```

  ```dart
  class ReportForm extends StatefulWidget {
    final String reportTargetId;  // 房间ID或订单ID
    final ReportType reportType;  // 'room' or 'order'

    @override
    State<ReportForm> createState() => _ReportFormState();
  }

  class _ReportFormState extends State<ReportForm> {
    String? selectedReason;
    String description = '';
    List<File> evidenceFiles = [];

    @override
    Widget build(BuildContext context) {
      return AlertDialog(
        title: Text('举报'),
        content: Column(
          children: [
            DropdownButton(
              items: [
                '骚扰与低俗内容',
                '实物送货等违规行为',
                '超时欠单',
                '其他违规',
              ].map((r) => DropdownMenuItem(value: r, child: Text(r))).toList(),
              onChanged: (value) => setState(() => selectedReason = value),
            ),
            TextField(
              maxLines: 3,
              onChanged: (v) => description = v,
              hintText: '请补充说明（可选）',
            ),
            ElevatedButton(
              onPressed: _pickEvidence,
              child: Text('上传证据 (${evidenceFiles.length}个文件)'),
            ),
          ],
        ),
        actions: [
          TextButton(onPressed: () => Navigator.pop(context), child: Text('取消')),
          FilledButton(onPressed: _submitReport, child: Text('提交举报')),
        ],
      );
    }

    Future<void> _pickEvidence() async {
      final result = await FilePicker.platform.pickFiles(
        allowMultiple: true,
        type: FileType.custom,
        allowedExtensions: ['jpg', 'png', 'pdf'],
      );
      if (result != null) {
        setState(() => evidenceFiles = result.paths.map((p) => File(p!)).toList());
      }
    }

    Future<void> _submitReport() async {
      try {
        await widget.api.submitReport(
          targetId: widget.reportTargetId,
          reportType: widget.reportType,
          reason: selectedReason!,
          description: description,
          evidenceUrls: evidenceFiles,
        );
        ScaffoldMessenger.of(context).showSnackBar(SnackBar(content: Text('举报已提交')));
        Navigator.pop(context);
      } catch (e) {
        ScaffoldMessenger.of(context).showSnackBar(SnackBar(content: Text(e.toString())));
      }
    }
  }
  ```
